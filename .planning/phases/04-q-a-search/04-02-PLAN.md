Create a service that performs hybrid search by combining Milvus vector similarity with Solr keyword matching using Reciprocal Rank Fusion (RRF).

Implement in html/modules/custom/social_ai_indexing/src/Service/HybridSearchService.php:

```php
<?php
declare(strict_types=1);

namespace Drupal\social_ai_indexing\Service;

use Drupal\Core\Session\AccountInterface;
use Drupal\search_api\Query\QueryInterface;
use Drupal\search_api\IndexInterface;

class HybridSearchService {
  
  protected PermissionFilterService $permissionFilter;
  protected $indexStorage;
  
  const RRF_K = 60; // Standard smoothing constant
  
  public function __construct(
    PermissionFilterService $permission_filter,
    $entity_type_manager
  ) {
    $this->permissionFilter = $permission_filter;
    $this->indexStorage = $entity_type_manager->getStorage('search_api_index');
  }
  
  public function search(string $query, AccountInterface $account, int $limit = 10): array {
    // 1. Semantic search via Milvus (social_posts index)
    $vectorResults = $this->vectorSearch($query, $account, $limit * 2);
    
    // 2. Keyword search via Solr
    $keywordResults = $this->keywordSearch($query, $account, $limit * 2);
    
    // 3. Merge with RRF
    $merged = $this->reciprocalRankFusion($vectorResults, $keywordResults);
    
    // 4. Post-retrieval access check (defense-in-depth)
    return $this->permissionFilter->filterResultsByAccess(
      array_slice($merged, 0, $limit),
      $account
    );
  }
  
  protected function reciprocalRankFusion(array $vectorResults, array $keywordResults): array {
    $scores = [];
    $merged = [];
    
    foreach ($vectorResults as $rank => $result) {
      $id = $this->getResultId($result);
      $scores[$id] = ($scores[$id] ?? 0) + (1 / ($rank + 1 + self::RRF_K));
      $merged[$id] = $result;
    }
    
    foreach ($keywordResults as $rank => $result) {
      $id = $this->getResultId($result);
      $scores[$id] = ($scores[$id] ?? 0) + (1 / ($rank + 1 + self::RRF_K));
      $merged[$id] = $result;
    }
    
    arsort($scores);
    return array_map(fn($id) => $merged[$id], array_keys($scores));
  }
  
  // Additional helper methods: vectorSearch(), keywordSearch(), getResultId()
}
```

Key implementation details:
- Use RRF_K = 60 (industry standard smoothing constant)
- Fetch 2x results from each source before RRF merge
- Apply PermissionFilterService to both queries AND post-retrieval
- Dedupe by entity ID (not Search API item ID)

Do NOT normalize scores between Milvus and Solr - use rank-based RRF instead.
Do NOT skip post-retrieval filtering - it's defense-in-depth security.

```php
<?php
declare(strict_types=1);

namespace Drupal\social_ai_indexing\Controller;

use Drupal\Core\Controller\ControllerBase;
use Drupal\social_ai_indexing\Service\HybridSearchService;
use Symfony\Component\DependencyInjection\ContainerInterface;
use Symfony\Component\HttpFoundation\JsonResponse;
use Symfony\Component\HttpFoundation\Request;

class HybridSearchController extends ControllerBase {
  
  protected HybridSearchService $hybridSearch;
  
  public static function create(ContainerInterface $container) {
    return new static($container->get('social_ai_indexing.hybrid_search'));
  }
  
  public function __construct(HybridSearchService $hybrid_search) {
    $this->hybridSearch = $hybrid_search;
  }
  
  public function search(Request $request): JsonResponse {
    $query = $request->query->get('q', '');
    $limit = (int) $request->query->get('limit', 10);
    
    if (strlen($query) < 2) {
      return new JsonResponse(['error' => 'Query too short'], 400);
    }
    
    $results = $this->hybridSearch->search($query, $this->currentUser(), $limit);
    
    return new JsonResponse([
      'query' => $query,
      'count' => count($results),
      'results' => array_map([$this, 'formatResult'], $results),
    ]);
  }
  
  protected function formatResult(array $result): array {
    return [
      'id' => $result['drupal_entity_id'] ?? $result['id'] ?? null,
      'title' => $result['citation_title'][0] ?? $result['title'][0] ?? '',
      'url' => $result['citation_url'][0] ?? '',
      'type' => $result['citation_type'][0] ?? '',
      'snippet' => substr($result['rendered_item'][0] ?? '', 0, 200),
    ];
  }
}
```
