/**
 * Implements hook_deepchat_settings().

 * Injects the current Group ID into DeepChat API requests so the RAG

function social_ai_indexing_deepchat_settings(array &$deepchat_settings) {
  // Detect current Group from route context.
  $group = \Drupal::service('social_group.current_group')->fromRunTimeContexts();
  if ($group) {
    // Add group_id to the contexts sent with each chatbot message.
    // The connect key is a PHP array at this point (not yet JSON-encoded).
    if (is_array($deepchat_settings['connect'] ?? NULL)) {
      $deepchat_settings['connect']['additionalBodyProps']['contexts']['group_id'] = (int) $group->id();
    }
  }
}

The Group context flows: DeepChat -> POST body contexts.group_id -> AiAssistantApiRunner::setContext() -> stored in runner. But when the RagTool executes a Search API query, the SearchQuerySubscriber needs to read the group_id from the runner's context and pass it to PermissionFilterService.

The challenge: SearchQuerySubscriber currently only reads `ai_search_scope_group_id` from query options, which RagTool doesn't set. The runner's context is stored on the `ai_assistant_api.runner` service.

Modify SearchQuerySubscriber to:
1. Inject the `ai_assistant_api.runner` service
2. In `onQueryPreExecute()`, if no explicit `ai_search_scope_group_id` query option is set, check the AI assistant runner's context for a `group_id` key
3. If found, use it as the scope group ID

Update the constructor and service definition:

```php
// In SearchQuerySubscriber:
use Drupal\ai_assistant_api\AiAssistantApiRunner;

class SearchQuerySubscriber implements EventSubscriberInterface {
  // Add new property:
  protected AiAssistantApiRunner $aiAssistantRunner;

  // Update constructor to accept the runner:
  public function __construct(
    PermissionFilterService $permission_filter,
    AccountInterface $current_user,
    AiAssistantApiRunner $ai_assistant_runner
  ) {
    $this->permissionFilter = $permission_filter;
    $this->currentUser = $current_user;
    $this->aiAssistantRunner = $ai_assistant_runner;
  }

  public function onQueryPreExecute(QueryPreExecuteEvent $event): void {
    $query = $event->getQuery();
    $index = $query->getIndex();

    if (!$this->isAiSearchIndex($index->id())) {
      return;
    }

    // Check for explicit scope from query options first.
    $scopeGroupId = $query->getOption('ai_search_scope_group_id');

    // If no explicit scope, check AI assistant runner context for group_id.
    if ($scopeGroupId === NULL) {
      $context = $this->aiAssistantRunner->getContext();
      if (!empty($context['group_id'])) {
        $scopeGroupId = (int) $context['group_id'];
      }
    }

    try {
      $this->permissionFilter->applyPermissionFilters(
        $query,
        $this->currentUser,
        $scopeGroupId !== NULL ? (int) $scopeGroupId : NULL
      );
    }
    catch (\Exception $e) {
      \Drupal::logger('social_ai_indexing')->warning(
        'Failed to apply permission filters to query on index @index: @message',
        [
          '@index' => $index->id(),
          '@message' => $e->getMessage(),
        ]
      );
    }
  }
}
```
