function social_ai_indexing_preprocess_block(array &$variables): void {
  if (($variables['elements']['#plugin_id'] ?? '') !== 'social_ai_related_content') {
    return;
  }
  $bundle = $variables['elements']['#configuration']['bundle'] ?? '';
  switch ($bundle) {
    case 'topic':
      $variables['subtitle'] = t('in the community');
      break;
    case 'event':
      $variables['subtitle'] = t('happening soon');
      break;
    default:
      $variables['subtitle'] = t('in the community');
  }
}
